    STRUCT
        RatioNumerator      : MC_LREAL         := 1;
        RatioDenumerator    : UINT             := 1;
        MasterSyncPosition  : MC_LREAL         := MC_INVALID;
        SlaveSyncPosition   : MC_LREAL         := MC_INVALID;
